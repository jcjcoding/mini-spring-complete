import java.util.ArrayDeque;
import java.util.Deque;

public class Main {
    public static void main(String[] args) {
        // 1. 初始化栈（Deque 接口指向 ArrayDeque 实现）
        Deque<Integer> stack = new ArrayDeque<>();

        // 2. 核心栈操作（LIFO 后进先出）
        stack.push(1);    // 入栈（栈顶）
        stack.push(2);
        int top = stack.peek(); // 查看栈顶（不移除，空栈返回 null）
        int pop = stack.pop();  // 出栈（移除并返回栈顶，空栈抛 NoSuchElementException）
        boolean isEmpty = stack.isEmpty(); // 判断栈空
        int size = stack.size(); // 栈大小
    }
}
